# Cursor runbook (exact commands)
# 1. Create venv and install
python -m venv .venv
.\.venv\Scripts\Activate
pip install --no-cache-dir -r requirements.txt

# 2. Validate canonical data
python engine/schema_validator.py data/hazcat_v1.1_equipment_types.json
python engine/schema_validator.py data/ruleset_v1.1_equipment_type_mappings.json
python engine/schema_validator.py data/wizard_mapping_equipment_types.json

# 3. Dry run
python engine/engine_core.py --mode dryrun --vectors data/unit_test_vectors_v1.json --output artifacts/dryrun

# 4. Seed expected fingerprints if first run
# Create data/expected_fingerprints_v1.json mapping id -> fingerprint from artifacts/dryrun/run_summary.json

# 5. CI check
python ci/test_vectors_runner.py --actual artifacts/dryrun/run_summary.json --expected data/expected_fingerprints_v1.json

# 6. Render human drafts
mkdir -p artifacts/dryrun/rendered
for d in artifacts/dryrun/*; do
  if [ -d "$d" ]; then
    for f in "$d"/*.json; do
      if [ -f "$f" ]; then
        python engine/render_engine.py "$f" templates/human_readable_template_markdown_v1.md rules/hazard_to_language_map_v1.json artifacts/dryrun/rendered
      fi
    done
  fi
done

# 7. Human draft tests
python ci/human_draft_tests.py --artifacts artifacts/dryrun/rendered --vectors data/unit_test_vectors_v1.json
